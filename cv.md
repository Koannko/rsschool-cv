### Code Example:
Write a program that will calculate and return the minimum of three numbers passed to it using the min function.
```
#include<stdio.h>
int min(int a, int b, int c){
    int m = a;
    if (b < a) {
        m = b;
    }
    
    if (c < a && c < b) {
        m = c;
    }
    
    return m;
}

int main(void){
    int x, y, z;
    scanf("%d%d%d", &x, &y, &z);
    
    int m = min(x, y, z);
    printf("%d", m);
    
    return 0;
}
```
